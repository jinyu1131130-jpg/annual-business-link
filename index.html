function submitNewMember() { const profile = checkProfile(); if (!profile) return; const branch = document.getElementById("newBranch").value.trim(); const name = document.getElementById("newName").value.trim(); const profession = document.getElementById("newProfession").value.trim(); const need = document.getElementById("newNeed").value.trim(); const lineId = document.getElementById("newLineId").value.trim(); const lineName = document.getElementById("newLineName").value.trim(); const lineLink = document.getElementById("newLineLink").value.trim(); const status = document.getElementById("addStatus"); if (!name) { alert("請填寫姓名"); return; } if (!profession) { alert("請填寫專業代表"); return; } if (!need) { alert("請填寫需要合作 / 引薦對象"); return; } const ok = confirm( "確認新增這位夥伴？\n\n" + "姓名：" + name + "\n" + "分會：" + (branch || "未填") + "\n" + "專業代表：" + profession ); if (!ok) return; status.style.display = "block"; status.innerHTML = "新增中..."; apiCall( { action: "addMember", branch: branch, name: name, profession: profession, need: need, lineId: lineId, lineName: lineName, lineLink: lineLink, creatorName: profile.recorderName, creatorBranch: profile.recorderBranch }, function(res) { if (res.ok) { status.innerHTML = "✅ 已新增到年會名單"; document.getElementById("newBranch").value = ""; document.getElementById("newName").value = ""; document.getElementById("newProfession").value = ""; document.getElementById("newNeed").value = ""; document.getElementById("newLineId").value = ""; document.getElementById("newLineName").value = ""; document.getElementById("newLineLink").value = ""; alert("✅ 已新增成功，現在可以用關鍵字搜尋到這位夥伴。"); } else { status.innerHTML = "新增失敗：" + escapeHtml(res.message); } }, function(error) { status.innerHTML = "新增失敗：" + escapeHtml(error.message); } ); }
